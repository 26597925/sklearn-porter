public static int {method_name}(double[] atts) {{
    if (atts.length != {n_features}) {{ return -1; }}

    {X}
    {y}

    int classIdx = -1;
    double minDist = Double.POSITIVE_INFINITY;
    double curDist;
    for (int i = 0; i < {n_templates}; i++) {{
        curDist = {class_name}.compDist(X[i], atts, {power});
        if (curDist < minDist) {{
            minDist = curDist;
            classIdx = y[i];
        }}
    }}
    return classIdx;
}}
