public static int {name}(double[] atts) {{
    if (atts.length != {n_features}) {{ return -1; }}

    {X}
    {y}

    int classIdx;
    double minDist = Double.INFINITY;
    for (int i = 0; i < templates.length; i++) {{
        if ({name}.compDist(X[i], atts) <= minDist) {{
            classIdx = y[i];
        }}
    }}
    return classIdx;
}}
